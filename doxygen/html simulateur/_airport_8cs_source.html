51 public List<Airport> RunAirport(List<Airport> airports)

70 }
71 else if (ClientList[i] is ObserverClient)

73 int observerIndex = GetAircraftIndex("observer");

75 }
76 else if (ClientList[i] is FireClient)
77 {
78 int fireIndex = GetAircraftIndex("fire");
79 SpecialDeparture(airports, fireIndex, i, true);

81 }
82
83 for (int i = 0; i < AircraftList.Count; i++)
84 {
85 if (AircraftList[i].GetState().GetType().Name.Equals("FlyingState"))
