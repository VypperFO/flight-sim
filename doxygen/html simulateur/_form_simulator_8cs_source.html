55 public Vector2 ConvertFromGPSToCoords(string gpsCoords)
56 {
57 string pattern = @"-?\d+";
58
59 MatchCollection matches = Regex.Matches(gpsCoords, pattern);
60
61 int[] numbers = new int[matches.Count];
62 for (int i = 0; i < matches.Count; i++)
63 {
64 numbers[i] = int.Parse(matches[i].Value);
65 }
66
67
68 double longitudeDegrees = numbers[0];
69 double longitudeMinutes = numbers[1];
70 double longitudeSeconds = numbers[2];
71
72 double latitudeDegrees = numbers[3];
73 double latitudeMinutes = numbers[4];
74 double latitudeSeconds = numbers[5];
75
76 int mapWidth = 600;
77 int mapHeight = 600;
78
79 double latitude = ConvertToDecimalDegrees(latitudeDegrees, latitudeMinutes, latitudeSeconds);
80 double longitude = ConvertToDecimalDegrees(longitudeDegrees, longitudeMinutes, longitudeSeconds);
81
82 Vector2 vector = new();
83 vector.X = (int)Math.Round(ConvertToX(longitude, mapWidth));
84 vector.Y = (int)Math.Round(ConvertToY(latitude, mapHeight));
85
86
87 return vector;
88 }

109 private double ConvertToX(double longitude, int mapWidth)
110 {
111 double x = (longitude + 180) * (mapWidth / 360.0);
112 return x;

121 private double ConvertToY(double latitude, int mapHeight)
122 {
123 double y = (90 - latitude) * (mapHeight / 180.0);
124 return y;
